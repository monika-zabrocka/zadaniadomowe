def convert_to_integers(mixed_list):
    converted_list = []
    
    for item in mixed_list:
        try:
            # zmieniamy na int
            converted_list.append(int(float(item)))
        except ValueError:
            #jesli nie dziala pomijamy
            print(f"Cannot convert '{item}' to an integer. Skipping this item.")
        except TypeError:
            
            print(f"Invalid type for conversion: {item}. Skipping this item.")
    
    return converted_list

# przyklad uzycia
mixed_data = [10, 3.14, '25', 'hello', [1, 2], '100', 8.9]
result = convert_to_integers(mixed_data)
print("Converted list:", result)
