def check_integers(a, b, c):
    if a > 0 and b > 0 and c > 0:
        return "All positive"
    elif a < 0 or b < 0 or c < 0:
        return "Contains negative"
    else:
        return "Mix of positive and negative"

print(check_integers(5, 10, 3))  #"All positive"
print(check_integers(-1, 5, 10))  #  "Contains negative"
print(check_integers(-5, 10, -2))  #  "Contains negative"
print(check_integers(5, -1, 3))   #  "Contains negative"
