DELIMITER $$

CREATE PROCEDURE GetStudentCourses(IN studentID INT)
BEGIN
    SELECT c.course_name, e.enrollment_date
    FROM enrollments e
    JOIN courses c ON e.course_id = c.course_id
    WHERE e.student_id = studentID;
END $$

DELIMITER ;



CALL GetStudentCourses(1);



DELIMITER $$

CREATE PROCEDURE UpdateEnrollmentDate(IN studentID INT, IN courseID INT, IN newDate DATE)
BEGIN
    UPDATE enrollments
    SET enrollment_date = newDate
    WHERE student_id = studentID AND course_id = courseID;
END $$

DELIMITER ;



CALL UpdateEnrollmentDate(1, 2, '2024-10-15');

DELIMITER $$

CREATE PROCEDURE RemoveStudentFromCourses(IN studentID INT)
BEGIN
    DELETE FROM enrollments
    WHERE student_id = studentID;
END $$

DELIMITER ;
